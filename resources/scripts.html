KQL Queries

Ingestion Health Check

KQL
// Check data freshness across all tables
union withsource=TableName *
| where TimeGenerated > ago(24h)
| summarize 
    EventCount = count(), 
    LastEvent = max(TimeGenerated),
    FirstEvent = min(TimeGenerated)
    by TableName
| extend HoursSinceLastEvent = datetime_diff('hour', now(), LastEvent)
| order by HoursSinceLastEvent desc

Cost Analysis by Table

KQL
// Monthly cost breakdown by data type
Usage
| where TimeGenerated > ago(30d)
| where IsBillable == true
| summarize 
    TotalGB = sum(Quantity) / 1000,
    EstimatedCost = sum(Quantity) / 1000 * 2.76
    by DataType
| order by TotalGB desc
| extend TotalGB = round(TotalGB, 2), EstimatedCost = round(EstimatedCost, 2)

Failed Sign-ins by User

KQL
SigninLogs
| where TimeGenerated > ago(24h)
| where ResultType != "0"
| summarize 
    FailedCount = count(),
    FailureCodes = make_set(ResultType),
    IPAddresses = make_set(IPAddress)
    by UserPrincipalName
| where FailedCount > 5
| order by FailedCount desc

User Activity Timeline

KQL
// Comprehensive user activity for investigation
let targetUser = "user@revflow.com";
let timeRange = 24h;
union SigninLogs, AuditLogs, OfficeActivity
| where TimeGenerated > ago(timeRange)
| where UserPrincipalName =~ targetUser or UserId =~ targetUser
| project TimeGenerated, 
    Source = $table,
    Activity = coalesce(OperationName, Operation, "Sign-in"),
    Result = coalesce(ResultType, ResultStatus, "Success"),
    Details = coalesce(ResultDescription, "")
| order by TimeGenerated desc

PowerShell Scripts

Check MDE Onboarding Status

PowerShell
# Check if device is onboarded to MDE
function Get-MDEOnboardingStatus {
    $regPath = "HKLM:\SOFTWARE\Microsoft\Windows Advanced Threat Protection\Status"
    
    try {
        $status = Get-ItemProperty -Path $regPath -ErrorAction Stop
        
        if ($status.OnboardingState -eq 1) {
            Write-Host "✓ Device is onboarded to MDE" -ForegroundColor Green
            Write-Host "  Organization ID: $($status.OrgId)"
        } else {
            Write-Host "✗ Device is NOT onboarded" -ForegroundColor Red
        }
    }
    catch {
        Write-Host "✗ MDE not installed or not configured" -ForegroundColor Red
    }
    
    # Check Sense service
    $sense = Get-Service -Name Sense -ErrorAction SilentlyContinue
    if ($sense) {
        Write-Host "  Sense Service: $($sense.Status)"
    }
}

Bulk DCR Association

PowerShell
# Associate DCR with multiple VMs
$dcrId = "/subscriptions/xxx/resourceGroups/rg-sentinel/providers/Microsoft.Insights/dataCollectionRules/DCR-WindowsSecurity"
$vms = Get-AzVM -ResourceGroupName "rg-servers"

foreach ($vm in $vms) {
    $associationName = "assoc-$($vm.Name)"
    
    New-AzDataCollectionRuleAssociation `
        -TargetResourceId $vm.Id `
        -AssociationName $associationName `
        -RuleId $dcrId
    
    Write-Host "Associated DCR with $($vm.Name)"
}

Enable Diagnostic Settings

Azure CLI
# Enable Azure Activity Log to Sentinel
az monitor diagnostic-settings subscription create \
  --name "SendToSentinel" \
  --subscription "YOUR_SUB_ID" \
  --workspace "/subscriptions/xxx/resourceGroups/rg-sentinel/providers/Microsoft.OperationalInsights/workspaces/sentinel-workspace" \
  --logs '[{"category": "Administrative", "enabled": true},{"category": "Security", "enabled": true},{"category": "Alert", "enabled": true}]'
