Operational Procedures

SOC Operations Operational Runbook

Day-to-day operational procedures for managing Microsoft Sentinel, MDE, and Purview DLP. These procedures ensure consistent operations across the security team.

1.2 Health Check Queries

KQL - Daily Health Check Dashboard
// Data connector health - last 24 hours
let ConnectorHealth = 
Heartbeat
| where TimeGenerated > ago(24h)
| summarize LastHeartbeat = max(TimeGenerated) by Computer, SourceComputerId
| extend Status = iff(LastHeartbeat < ago(30m), "Unhealthy", "Healthy");

// Log ingestion volume trend
let IngestionTrend =
Usage
| where TimeGenerated > ago(7d)
| summarize DailyGB = sum(Quantity) / 1024 by bin(TimeGenerated, 1d), DataType
| order by TimeGenerated desc;

// Incident metrics
let IncidentMetrics =
SecurityIncident
| where TimeGenerated > ago(24h)
| summarize 
    TotalIncidents = count(),
    CriticalCount = countif(Severity == "High"),
    OpenCount = countif(Status == "New" or Status == "Active"),
    ClosedCount = countif(Status == "Closed"),
    MTTR = avg(datetime_diff('hour', ClosedTime, CreatedTime));

// Alert rule health
let RuleHealth =
SentinelHealth
| where TimeGenerated > ago(24h)
| where SentinelResourceType == "Analytic Rule"
| summarize 
    TotalRuns = count(),
    FailedRuns = countif(Status == "Failure"),
    SuccessRate = round(100.0 * countif(Status == "Success") / count(), 2)
    by SentinelResourceName
| where FailedRuns > 0;

// Output all metrics
ConnectorHealth | where Status == "Unhealthy"

2.1 Connector Health Monitoring

KQL - Connector Last Data Received
// Check when each table last received data
union withsource=TableName *
| summarize LastData = max(TimeGenerated) by TableName
| extend HoursSinceData = datetime_diff('hour', now(), LastData)
| where HoursSinceData > 1
| order by HoursSinceData desc

// Specific connector checks
// Windows Security Events
SecurityEvent
| summarize LastEvent = max(TimeGenerated), EventCount = count() by Computer
| where LastEvent < ago(1h)
| order by LastEvent asc

// Palo Alto Firewall
CommonSecurityLog
| where DeviceVendor == "Palo Alto Networks"
| summarize LastLog = max(TimeGenerated), LogCount = count()
| extend MinutesSinceLog = datetime_diff('minute', now(), LastLog)

// MDE Connector
DeviceInfo
| summarize LastSeen = max(TimeGenerated) by DeviceName
| where LastSeen < ago(24h)
| order by LastSeen asc

3. Analytics Rule Management

3.1 Rule Health Check

KQL - Analytics Rule Execution Status
// Check rule execution health
SentinelHealth
| where TimeGenerated > ago(24h)
| where SentinelResourceType == "Analytic Rule"
| summarize 
    LastRun = max(TimeGenerated),
    TotalRuns = count(),
    SuccessCount = countif(Status == "Success"),
    FailureCount = countif(Status == "Failure"),
    SuccessRate = round(100.0 * countif(Status == "Success") / count(), 2)
    by SentinelResourceName
| extend 
    HealthStatus = case(
        SuccessRate == 100, "Healthy",
        SuccessRate >= 90, "Warning",
        "Critical"
    )
| where HealthStatus != "Healthy"
| order by SuccessRate asc

// Get failure details
SentinelHealth
| where TimeGenerated > ago(24h)
| where Status == "Failure"
| project TimeGenerated, SentinelResourceName, Description, ExtendedProperties
| order by TimeGenerated desc

3.2 Common Rule Failures

Error TypeCauseResolution
"Semantic error" KQL syntax error, missing column Edit rule, fix query syntax, verify table schema
"Query timeout" Query too complex, large time range Optimize query, reduce lookback period, add filters
"No data" Data source not ingesting, filter too restrictive Check connector health, broaden filters
"Throttled" Too many rules running simultaneously Stagger rule schedules, reduce frequency

3.3 Tuning False Positives

KQL - Identify Noisy Rules
// Find rules generating most false positives
SecurityIncident
| where TimeGenerated > ago(30d)
| where Classification == "FalsePositive" or Classification == "BenignPositive"
| summarize FalsePositiveCount = count() by AlertProductNames
| order by FalsePositiveCount desc
| take 10

// Detailed false positive analysis for a specific rule
let RuleName = "Suspicious PowerShell Execution";
SecurityIncident
| where TimeGenerated > ago(30d)
| where Title has RuleName
| where Classification == "FalsePositive"
| extend Entities = parse_json(Entities)
| mv-expand Entities
| where Entities.Type == "host" or Entities.Type == "account"
| summarize 
    FPCount = count(),
    ExampleTitles = make_set(Title, 3)
    by tostring(Entities.Name)
| order by FPCount desc

4.2 Threat & Vulnerability Management

KQL - TVM Priority Vulnerabilities
// Critical vulnerabilities by device
DeviceTvmSoftwareVulnerabilities
| where VulnerabilitySeverityLevel == "Critical"
| summarize 
    CriticalVulns = dcount(CveId),
    Vulnerabilities = make_set(CveId, 10)
    by DeviceName
| order by CriticalVulns desc
| take 20

// Vulnerabilities with known exploits
DeviceTvmSoftwareVulnerabilities
| where IsExploitAvailable == true
| summarize 
    AffectedDevices = dcount(DeviceName),
    Devices = make_set(DeviceName, 10)
    by CveId, VulnerabilitySeverityLevel
| order by AffectedDevices desc

// Software requiring patching
DeviceTvmSoftwareVulnerabilities
| summarize VulnCount = count() by SoftwareName, SoftwareVersion
| where VulnCount > 5
| order by VulnCount desc

| summarize 
    BlockCount = countif(ActionType endswith "Blocked"),
    AuditCount = countif(ActionType endswith "Audited"),

| order by BlockCount desc

DeviceEvents
| where TimeGenerated > ago(7d)

5. DLP Operations

5.1 Policy Violation Review

KQL - DLP Violation Summary
// Daily DLP violation summary
PurviewDLPLogs
| where TimeGenerated > ago(24h)
| summarize 
    TotalViolations = count(),
    UniqueUsers = dcount(User),
    BlockedActions = countif(ActionsTaken has "Block"),
    WarnActions = countif(ActionsTaken has "Notify")
    by PolicyName
| order by TotalViolations desc

// High-risk violations requiring immediate review
PurviewDLPLogs
| where TimeGenerated > ago(24h)
| where SensitiveInfoType has_any ("Credit Card", "SSN", "Bank Account", "Protected Health Information")
| where ActionsTaken has "Block"
| project 
    TimeGenerated,
    User,
    PolicyName,
    SensitiveInfoType,
    Workload,
    ItemPath,
    Recipients
| order by TimeGenerated desc

// Repeat offenders
PurviewDLPLogs
| where TimeGenerated > ago(30d)
| summarize ViolationCount = count() by User
| where ViolationCount > 10
| order by ViolationCount desc

5.2 DLP Policy Effectiveness

KQL - Policy Effectiveness Metrics
// Policy trigger trends (weekly)
PurviewDLPLogs
| where TimeGenerated > ago(30d)
| summarize ViolationCount = count() by PolicyName, bin(TimeGenerated, 1w)
| order by TimeGenerated desc, ViolationCount desc

// Action distribution by policy
PurviewDLPLogs
| where TimeGenerated > ago(30d)
| extend Action = case(
    ActionsTaken has "Block", "Blocked",
    ActionsTaken has "Notify", "Warned",
    "Logged"
)
| summarize Count = count() by PolicyName, Action
| order by PolicyName, Count desc

6. Reporting Procedures

6.1 Weekly Security Report

KQL - Weekly Metrics Query
// Weekly security metrics summary
let StartDate = startofweek(ago(7d));
let EndDate = startofweek(now());

// Incident metrics
let IncidentMetrics = SecurityIncident
| where TimeGenerated between (StartDate .. EndDate)
| summarize 
    TotalIncidents = count(),
    CriticalHigh = countif(Severity in ("High", "Critical")),
    Resolved = countif(Status == "Closed"),
    AvgMTTR = avg(datetime_diff('hour', ClosedTime, CreatedTime))
| extend ResolvedPct = round(100.0 * Resolved / TotalIncidents, 1);

// Alert metrics
let AlertMetrics = SecurityAlert
| where TimeGenerated between (StartDate .. EndDate)
| summarize 
    TotalAlerts = count(),
    HighSeverity = countif(AlertSeverity == "High")
| extend AlertsPerDay = round(1.0 * TotalAlerts / 7, 0);

// MDE metrics
let MDEMetrics = DeviceInfo
| summarize 
    TotalDevices = dcount(DeviceId),
    OnboardedThisWeek = dcountif(DeviceId, FirstSeen > StartDate);

// Output report
print "Weekly Security Report"
| extend WeekStart = StartDate, WeekEnd = EndDate
| extend 
    Incidents_Total = toscalar(IncidentMetrics | project TotalIncidents),
    Incidents_CriticalHigh = toscalar(IncidentMetrics | project CriticalHigh),
    Incidents_ResolvedPct = toscalar(IncidentMetrics | project ResolvedPct),
    Incidents_AvgMTTR = toscalar(IncidentMetrics | project AvgMTTR),
    Alerts_Total = toscalar(AlertMetrics | project TotalAlerts),
    Alerts_PerDay = toscalar(AlertMetrics | project AlertsPerDay),
    Devices_Total = toscalar(MDEMetrics | project TotalDevices),
    Devices_NewOnboarded = toscalar(MDEMetrics | project OnboardedThisWeek)

7. Change Management

7.1 Analytics Rule Changes

  1. Document change request with justification
  2. Test in dev/test workspace if available
  3. Peer review query changes
  4. Schedule change window (avoid peak hours)
  5. Implement change with rollback plan
  6. Monitor for 24 hours post-change
  7. Document outcome in change log

7.2 MDE Policy Changes

  1. Create change request with impact assessment
  2. Test on pilot group (10-50 devices)
  3. Monitor pilot for 1 week
  4. Review for false positives or user impact
  5. Expand to production in waves (25% → 50% → 100%)
  6. Document exceptions if needed

7.3 DLP Policy Changes

  1. Business stakeholder approval required
  2. Legal review for new sensitive info types
  3. Deploy in test mode first
  4. Review simulated matches for 2 weeks
  5. Enable enforcement after validation
  6. User communication before enforcement
