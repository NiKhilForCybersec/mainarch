Common Investigation Queries

KQL - User Activity Timeline
// Get all activity for a user in last 24 hours
let targetUser = "user@revflow.com";
union SigninLogs, AuditLogs, DeviceLogonEvents
| where TimeGenerated > ago(24h)
| where UserPrincipalName =~ targetUser 
    or TargetUserName =~ targetUser
    or AccountUpn =~ targetUser
| project TimeGenerated, 
    Source = $table, 
    Activity = coalesce(OperationName, ActionType),
    Details = coalesce(ResultDescription, AdditionalDetails)
| order by TimeGenerated desc
KQL - IP Reputation Check
// Check if IP appears in threat intel
let suspiciousIP = "1.2.3.4";
ThreatIntelligenceIndicator
| where NetworkIP == suspiciousIP or NetworkSourceIP == suspiciousIP
| project TimeGenerated, ThreatType, Description, ConfidenceScore
